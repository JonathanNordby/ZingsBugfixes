character_event = {
	id = wildling.100
	picture = GFX_evt_death
	
	desc = {
		text = "EVTDESCwildling.100"
		trigger = { 
			trait = wildling
			NOT = { has_character_flag = wildling_defeat_realm_fracture }
		}
	}
	desc = {
		text = "EVTDESCwildling.100B"
		trigger = { 
			NOT = { trait = wildling } 
			NOT = { has_character_flag = wildling_defeat_realm_fracture }
		}
	}
	desc = {
		text = "EVTDESCwildling.100C"
		trigger = { has_character_flag = wildling_defeat_realm_fracture }
	}
	
	is_triggered_only = yes
	only_rulers = yes

	trigger = {
		government = wildling_feudal_government #also applies to shadowmen
		OR = {
			AND = { #High rank, weak heir
				higher_tier_than = COUNT
				current_heir = {
					OR = {
						is_female = yes
						NOT = { age = 16 }
						NOR = {
							prestige = 500
							martial = 10
							trait = master_warrior
							diplomacy = 10
						}
					}	
				}
			}
			AND = { #Non free folk deserter with non-wildling heir
				trait = deserter
				NOT = { current_heir = { trait = wildling } }
			}
			has_character_flag = wildling_defeat_realm_fracture
		}		
	}
	
	immediate = {
		#Mark current liege
		if = { 
			limit = { independent = no }
			hidden_tooltip = { 
				liege = { save_event_target_as = wildling_liege } 
				liege_before_war = { save_event_target_as = wildling_liege } 
			} 
		}
		
		#Dserters lose all land if no wildling heir
		if = {
			limit = {
				trait = deserter
				NOT = { current_heir = { trait = wildling } }
			}
			random_courtier = {
				limit = {
					NOT = { character = ROOT }
					trait = wildling
					is_ruler = no
					is_female = no
					age = 16				
					prisoner = no
					NOT = { trait = incapable }
					NOT = { trait = maester }
					NOT = { trait = nightswatch }
					NOT = { trait = kingsguard }
				}
				preferred_limit = {
					OR = {
						prestige = 500
						martial = 10
						trait = master_warrior
						diplomacy = 10
					}
				}
				preferred_limit = {
					martial = 8
				}
				save_event_target_as = wildling_usurper 
			}
			if = {
				limit = { NOT = { event_target:wildling_usurper = { always = yes } } }
				capital_scope = {
					if = {
						limit = { wildling_culture_trigger = yes }
						create_random_soldier = {
							random_traits = yes
							dynasty = actually_culture
							religion = THIS
							culture = THIS
							female = no
						}
					}
					else = {
						create_random_soldier = {
							random_traits = yes
							dynasty = actually_culture
							religion = beyond_wall_old_gods
							culture = wildling
							female = no
						}
					}
					new_character = {
						character_event = { id = wildling.1 } #check for wildling trait
						change_martial = 2
						change_diplomacy = 2
						random_list = {
							30 = { add_trait = skilled_warrior }
							40 = { add_trait = trained_warrior }
							20 = { add_trait = poor_warrior }
						}
						save_event_target_as = wildling_usurper
					}
				}				
			}
		}
		#if spare county held, chance of courtier usurping capital and duchies
		else_if = {
			limit = {
				is_feudal = yes
				any_demesne_title = {
					is_landless_type_title = no
					tier = DUKE
				}
				num_of_count_titles = 2
			}
			random_courtier = {
				limit = {
					NOT = { character = ROOT }
					trait = wildling
					is_ruler = no
					is_female = no
					age = 16
					OR = {
						prestige = 500
